/**
 * A simple example of a Java class demonstrating cloud infrastructure concepts
 */
public class CloudInfrastructureExample {
    private final String region;
    private final int instanceCount;
    private Map instances;

    public CloudInfrastructureExample(String region, int instanceCount) {
        this.region = region;
        this.instanceCount = instanceCount;
        this.instances = new HashMap<>();
    }

    /**
     * Demonstrates auto-scaling based on load
     */
    public void scaleInstances(double currentLoad) {
        if (currentLoad > 0.8 && instances.size() < instanceCount) {
            // Scale up when load is above 80%
            Instance newInstance = createInstance();
            instances.put(newInstance.getId(), newInstance);
            System.out.println("Scaled up: New instance created in " + region);
        } else if (currentLoad < 0.2 && instances.size() > 1) {
            // Scale down when load is below 20%
            String instanceId = instances.keySet().iterator().next();
            instances.remove(instanceId);
            System.out.println("Scaled down: Instance terminated in " + region);
        }
    }

    private Instance createInstance() {
        return new Instance(UUID.randomUUID().toString(), region);
    }

    private static class Instance {
        private final String id;
        private final String region;

        public Instance(String id, String region) {
            this.id = id;
            this.region = region;
        }

        public String getId() {
            return id;
        }
    }
}
